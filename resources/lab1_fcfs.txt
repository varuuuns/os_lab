#include<bits/stdc++.h>
using namespace std;

int search(vector<int> arr,int x){
    for(int i=0;i<arr.size();i++){
        if(arr[i]==x) return i;
    }
}

void fcfs(vector<int> arrival,vector<int> burst){
    int n=burst.size();
    vector<int> temp1=arrival;
    vector<int> temp2=burst;
    vector<int> process;
    for(int i=0;i<n;i++) process.push_back(i);
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            if(arrival[i]>arrival[j]){
                swap(temp2[i],temp2[j]);
                swap(temp1[i],temp1[j]);
                swap(process[i],process[j]);
            }
        }
    }
    for(int i=0;i<n;i++){
        cout<<temp1[i]<<" ";
        cout<<endl;
    }
    for(int i=0;i<n;i++){
        cout<<temp2[i]<<" ";
        cout<<endl;
    } 
    vector<int> ct;
    vector<int> tat;
    vector<int> wt;
    int temp=0;
    for(int i=0;i<n;i++){
        ct.push_back(temp2[i]+temp);
        tat.push_back(ct[i]-temp1[i]);
        wt.push_back(tat[i]-temp2[i]);
        temp+=temp2[i];
    }
    for(int i=0;i<n;i++){
        cout<<ct[i]<<" ";
    }

    // for(int i=1;i<=n;i++){

    //     cout<<"Turn Around Time "<<tat[t]<<" ";
    //     cout<<"Waiting Time "<<wt[t]<<endl;
    // }
}

int main(){
    int n;
    cout<<"Enter number of processess: ";
    cin>>n;
    vector<int> at(n);
    vector<int> bt(n);
    for(int i=1;i<=n;i++){
        cout<<"Enter Arrival Time of Process "<<i<<": ";
        cin>>at[i];
        cout<<"Burst Arrival Time of Process "<<i<<": ";
